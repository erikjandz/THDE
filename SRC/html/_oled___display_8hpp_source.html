110  // RTOS main
111  void main() override
112  {
113  for(;;)
114  {
115  switch(_state)
116  {
117  case State::IDLE:
118  wait(_displayFlag);
119  _state = State::ACTIVE;
120  break;
121 
122  case State::ACTIVE:
123  for(auto & s : _displayPool.read())
124  {
125  if(s.initialized)
126  {
127  if(s.isChar)
128  {
129  _textWriter << s.c;
130  }
131  else
132  {
133  _textWriter << s.i;
134  }
135  }
136  else
137  {
138  break;
139  }
140  }
141 
142  _state = State::IDLE;
143  break;
144  }
145  }
146  }
